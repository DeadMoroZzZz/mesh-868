<script>

    // Настройки shiftr.io

    const MQTT_CONFIG = {

        host: "wss://broker.shiftr.io",

        user: "ВАШ_TOKEN_KEY", 

        pass: "ВАШ_TOKEN_SECRET"

    };


    const SERVICE_UUID = "6ba1b088-1b4d-4d2a-942b-bc6830501f28";

    const TO_RADIO_UUID = "63820054-0617-4861-9c60-496350f3dd90";

    const FROM_RADIO_UUID = "2086938a-446a-466d-8a21-4ba29c716246";


    let toChar, mqttClient = null, cryptoKey = "", authFails = 0, currentTopic = "", qrcode = null;

    let savedTopicName = "", savedTopicPass = "";


    // При загрузке проверяем URL на наличие конфига

    window.onload = () => {

        qrcode = new QRCode(document.getElementById("qrcode"), { text: "https://t.me/just_be_freee", width: 128, height: 128 });

        const params = new URLSearchParams(window.location.search);

        if(params.has('t')) document.getElementById('mqtt-topic').value = params.get('t');

        if(params.has('p')) document.getElementById('topic-pass').value = params.get('p');

    };


    function toggleInfoModal() {

        const m = document.getElementById('info-modal');

        if (m.style.display !== 'flex') {

            document.getElementById('modal-title').innerText = "О ПРОГРАММЕ";

            document.getElementById('modal-desc').innerHTML = "Автор: <b>Захарченко Виктор</b>";

            document.getElementById('modal-footer').innerText = "t.me/just_be_freee";

            qrcode.clear(); qrcode.makeCode("https://t.me/just_be_freee");

        }

        m.style.display = (m.style.display === 'flex') ? 'none' : 'flex';

    }


    function shareConfig() {

        const configUrl = `${window.location.href.split('?')[0]}?t=${encodeURIComponent(savedTopicName)}&p=${encodeURIComponent(savedTopicPass)}`;

        document.getElementById('modal-title').innerText = "НАСТРОЙКИ ГРУППЫ";

        document.getElementById('modal-desc').innerText = "Сканируйте для входа в канал";

        document.getElementById('modal-footer').innerText = "Топик и Пароль переданы";

        qrcode.clear(); qrcode.makeCode(configUrl);

        document.getElementById('info-modal').style.display = 'flex';

    }


    const crypt = (t, k) => t.split('').map((c, i) => String.fromCharCode(c.charCodeAt(0) ^ k.charCodeAt(i % k.length))).join('');


    async function getHashedTopic(name, pass) {

        const msgUint8 = new TextEncoder().encode(name.trim().toUpperCase() + pass);

        const hashBuffer = await crypto.subtle.digest('SHA-256', msgUint8);

        const hashArray = Array.from(new Uint8Array(hashBuffer));

        return "m868/" + hashArray.map(b => b.toString(16).padStart(2, '0')).join('').substring(0, 12);

    }


    async function handleConnect() {

        const code = document.getElementById('access-code').value.trim();

        savedTopicName = document.getElementById('mqtt-topic').value.trim() || "GENERAL";

        savedTopicPass = document.getElementById('topic-pass').value.trim();

        cryptoKey = document.getElementById('crypto-key').value.trim();


        let sum = 0; for(let i=0; i<code.length; i++) sum += code.charCodeAt(i);

        if (!/^(?=.*[a-zA-Z])(?=.*\d).{6,}$/.test(code) || sum % 7 !== 0) {

            authFails++; if(authFails >= 3) { localStorage.clear(); location.reload(); }

            return alert("Неверный код доступа!");

        }


        currentTopic = await getHashedTopic(savedTopicName, savedTopicPass);


        try {

            const dev = await navigator.bluetooth.requestDevice({ filters:[{namePrefix:'Meshtastic'}], optionalServices:[SERVICE_UUID] });

            const srv = await dev.gatt.connect();

            const svc = await srv.getPrimaryService(SERVICE_UUID);

            toChar = await svc.getCharacteristic(TO_RADIO_UUID);

            const from = await svc.getCharacteristic(FROM_RADIO_UUID);

            await from.startNotifications();

            from.addEventListener('characteristicvaluechanged', handleRadioIn);

            

            initMQTT();

            document.getElementById('screen-auth').classList.remove('active');

            document.getElementById('screen-nick').classList.add('active');

        } catch(e) { alert("BT/Connection Error: " + e.message); }

    }


    function initMQTT() {

        if (!navigator.onLine) return;

        mqttClient = mqtt.connect(MQTT_CONFIG.host, { username: MQTT_CONFIG.user, password: MQTT_CONFIG.pass, clientId: 'v_'+Math.random().toString(16).substr(2, 4) });

        mqttClient.on('connect', () => {

            const s = document.getElementById('net-status');

            s.innerText = "ONLINE"; s.style.color = "var(--prim)";

            mqttClient.subscribe(currentTopic);

        });

        mqttClient.on('message', (topic, msg) => {

            const dec = crypt(msg.toString(), cryptoKey);

            if(!isDup(dec)) { addUI(dec, 'other'); sendRaw(dec); saveMsg(dec, 'other'); }

        });

    }


    function startSession() {

        const n = document.getElementById('user-nick').value.trim();

        if(!n) return;

        localStorage.setItem('m_nick', n);

        document.getElementById('display-name').innerText = n.toUpperCase();

        document.getElementById('screen-nick').classList.remove('active');

        document.getElementById('screen-chat').classList.add('active');

        loadHist();

    }


    async function sendMsg() {

        const i = document.getElementById('chat-input'), n = localStorage.getItem('m_nick');

        if(!i.value.trim()) return;

        const txt = `${n}: ${i.value.trim()}`;

        addUI(txt, 'self'); saveMsg(txt, 'self');

        await sendRaw(txt);

        if(mqttClient && mqttClient.connected) mqttClient.publish(currentTopic, crypt(txt, cryptoKey));

        i.value = "";

    }


    function handleRadioIn(e) {

        const data = new Uint8Array(e.target.value.buffer);

        if(data[0] === 0x08) document.getElementById('batt-info').innerText = `🔋 ${data[1]}%`;

        const t = new TextDecoder().decode(data).replace(/[^\x20-\x7EА-Яа-яЁё:/,;=.-]/g, '');

        if(t.includes(':') && !isDup(t)) { 

            addUI(t, 'other'); saveMsg(t, 'other'); 

            if(mqttClient && mqttClient.connected) mqttClient.publish(currentTopic, crypt(t, cryptoKey));

        }

    }


    function addUI(t, cl) {

        const b = document.getElementById('messages'), e = document.createElement('div');

        e.className = `msg ${cl}`; e.innerText = t;

        b.appendChild(e); b.scrollTop = b.scrollHeight;

    }


    let lastMsgs = [];

    function isDup(t) { if(lastMsgs.includes(t)) return true; lastMsgs.push(t); if(lastMsgs.length>20) lastMsgs.shift(); return false; }

    function saveMsg(t, type) { 

        let h = JSON.parse(localStorage.getItem('h_v7')||'[]');

        h.push({t: crypt(t, cryptoKey), type});

        if(h.length>50) h.shift();

        localStorage.setItem('h_v7', JSON.stringify(h));

    }

    function loadHist() { JSON.parse(localStorage.getItem('h_v7')||'[]').forEach(m => addUI(crypt(m.t, cryptoKey), m.type)); }

    async function sendRaw(t) { try { await toChar.writeValue(new TextEncoder().encode(t)); } catch(e){} }

    async function triggerSOS() { const n = localStorage.getItem('m_nick'); await sendRaw(`SOS:${n}`); addUI("🚨 SOS ОТПРАВЛЕН", "self"); }

    function sendLocation() { navigator.geolocation.getCurrentPosition(async p => {

        const txt = `LOC:${localStorage.getItem('m_nick')}:${p.coords.latitude.toFixed(6)}:${p.coords.longitude.toFixed(6)}`;

        addUI("📍 Координаты отправлены", "self"); await sendRaw(txt);

    }); }

    document.getElementById('chat-input').onkeypress = e => { if(e.key==='Enter') sendMsg(); };
